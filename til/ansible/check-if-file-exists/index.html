Check if File Exists
3 Jan 2023 • 1 min

tasks:
  - name: Check if path exists
    stat:
	    path: "/path/to/file"
	register: result

  - name: Do something if path exists
    command: ...
    when: result.stat.exists

  - name: Do something else if path does not exists
    command: ...
    when: not result.stat.exists
